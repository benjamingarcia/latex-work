\section*{\center{Bilan proportionnalité}}




\bigskip

\begin{center}
\begin{tabular}{|m{95mm}|m{95mm}|}
\hline


\textbf{Ce que je dois savoir} & \textbf{Ce que je dois savoir faire} \\

\hline

\enskip


\begin{itemize}
  \item [$\bullet$] Je sais reconnaître un tableau de proportionnalité.
  
  
  \item [$\bullet$] Je connais l'égalité des produits en croix.
  
  \item [$\bullet$] Je connais la propriété sur la représentation graphique
  d'une situation de proportionnalité.
  
  \item [$\bullet$] Je connais les propriétés et formules sur la vitesse
  moyenne.

 \end{itemize}

&


\begin{itemize}
  \item[$\bullet$] Je sais trouver le coefficient de proportionnalité.


  \item[$\bullet$] Je sais effectuer un calcul en utilisant la règle des
  produits en croix.

  
      
  \item[$\bullet$] Je sais déterminer une quatrième proportionnelle.
  
  \item[$\bullet$] Je sais appliquer un pourcentage.
  
  \item[$\bullet$] Je sais calculer un pourcentage.
  
  \item[$\bullet$] Je sais calculer un pourcentage ``groupé''.
  
  \item[$\bullet$] Je sais convertir des km/h en m/s et inversément.
  
  \item[$\bullet$] Je sais résoudre des problèmes de vitesse.
 
\end{itemize} \\

\hline

\end{tabular}
\end{center}
