\subsection*{1) droites parallèles et perpendiculaires}

\begin{center}
\begin{tabular}{|p{7cm}|p{12cm}|}
\hline

\textbf{FIGURES} & \textbf{PROPRIETES}\\


\hline
  
  & \\ 


  & \begin{itemize}
     \item [$\bullet$] Si deux droites sont parallèles à une même troisième
     alors elles sont parallèles entre elles.
    
    \bigskip
    
    
     \item [$\bullet$] Si deux droites sont parallèles, toute
     perpendiculaire à l'une est perpendiculaire à l'autre.
     
     \bigskip
     
     
     \item [$\bullet$] Si deux droites sont perpendiculaires à une même droite,
     alors elles sont parallèles.
\end{itemize} \\ 

 \hline

 & \\

 & \ul{Théorème de Thalès:} 


Soient $(AA')$ et $(BB')$ deux droites
 sécantes en un point $O$. Si les droites $(AB)$ et $(A'B')$ sont parallèles,
 alors $\dfrac{OA}{OA'}=\dfrac{OB}{OB'}=\dfrac{AB}{A'B'}$.
 \\
  & \\ 
  
 \hline 
 
 & \\

 & \ul{Réciproque du théorème de Thalès:} 


Soient $(AA')$ et $(BB')$ deux droites
 sécantes en un point $O$. Si les points $O$, $A$, $A'$ et $O$, $B$, $B'$ sont
 alignés dans le même ordre et si $\dfrac{OA}{OA'}=\dfrac{OB}{OB'}$, alors les
 droites $(AB)$ et $(A'B')$ sont parallèles et de plus
 $\dfrac{OA}{OA'}=\dfrac{AB}{A'B'}$.\\

 & \\
 \hline
\end{tabular}
\end{center}

\subsection*{2) Propriétés des quadrilatères}

\begin{center}
\begin{tabular}{|p{7cm}|p{12cm}|}
\hline

\textbf{FIGURES} & \textbf{PROPRIETES}\\
\hline

  & \\ 
 & \ul{Propriétés caractéristiques d'un parallélogramme:}
 
 
 \begin{itemize}
   \item [$\bullet$] $ABCD$ est un parallélogramme si et seulement si $[AC]$ et
   $[BD]$ ont le même milieu.
   
   \enskip
   
   \item [$\bullet$] $ABCD$ est un parallélogramme si et seulement si $(AB)$
   parallèle à $(CD)$ et $(AD)$ parallèle à $(BC)$.
   
   \enskip
   
   \item [$\bullet$] $ABCD$ est un parallélogramme si et seulement si $AB=CD$
   et $AD=BC$.
   
   \enskip
   
   \item [$\bullet$] $ABCD$ est un parallélogramme si et seulement si $(AB)$
   parallèle à $(CD)$ et $AB=CD$.
  
  \enskip
  
  \item [$\bullet$] $ABCD$ est un parallélogramme si et seulement si
  $\widehat{DAB}=\widehat{BCD}$ et $\widehat{ABC}=\widehat{CDA}$.
 \end{itemize} \\
 
  & \\
  \hline
  
 \end{tabular} 
\end{center}

\begin{center}
\begin{tabular}{|p{7cm}|p{12cm}|}
\hline

\textbf{FIGURES} & \textbf{PROPRIETES}\\
\hline

  & \\ 
 & \ul{Propriétés caractéristiques d'un losange:}
 
 
 \begin{itemize}
   \item [$\bullet$] $ABCD$ est un losange si et seulement si $ABCD$ est un
   parallélogramme dont deux côtés consécutifs sont égaux
   
   \enskip
   
   \item [$\bullet$] $ABCD$ est un losange si et seulement si $ABCD$ est un
   parallélogramme  tel que $(AC)$ et $(BD)$ perpendiculaires.
 \end{itemize}\\
  & \\
  \hline
   & \\
  & \ul{Propriétés caractéristiques d'un rectangle:}
  
  \begin{itemize}
    \item [$\bullet$] $ABCD$ est un rectangle si et seulement si $ABCD$ est un
   parallélogramme dont deux côtés consécutifs sont perpendiculaires.
   \item [$\bullet$] $ABCD$ est un rectangle si et seulement si $ABCD$ est un
   parallélogramme tel que $AC=BD$.  
  \end{itemize}\\
   & \\
   \hline
   & \\
   & \ul{Propriété caractéristique d'un carré:}
   
   $\bullet$ $ABCD$ est un carré si et seulement si $ABCD$ est un losange et un
   rectangle.\\
   &  \\
   \hline
 \end{tabular}

\subsection*{3) Droites remarquables d'un triangle $ABC$}
\begin{center}
\begin{tabular}{|p{7cm}|p{12cm}|}
\hline

\textbf{FIGURES} & \textbf{PROPRIETES}\\
\hline

  & \\ 
 & \ul{Médianes:}
 
 
 \begin{itemize}
     \item [$\bullet$] \ul{définition:} Une médiane est une droite passant par
     un sommet et par le mileu du côté opposé.
     
     \enskip
     
     \item [$\bullet$] Les trois médianes d'un triangle sont concourantes en un
     point $G$ appelé centre de gravité. De plus, si $A'$ est le mileu du côté
     $[BC]$, on a: $AG=\dfrac{2}{3}AA'$.
     
      \enskip
      
       \item [$\bullet$] la médiane partage le triangle en deux triangles de
       même aire: aire($AA'B$)=aire($AA'C$).
 \end{itemize} \\
  \hline
 & \\
 & \ul{Hauteurs:}

 
 \begin{itemize}
     \item [$\bullet$] \ul{définition:} Une hauteur est une droite passant par
     un sommet et qui est perpendiculaire au côté opposé.
     
     \enskip
     
     \item [$\bullet$] Les trois hauteurs d'un triangle sont concourantes en un
     point $H$ appelé orthocentre.
     
     \enskip
     
     \item [$\bullet$] Soit $K$ le point d'intersection de la hauteur issue de $A$ et du côté
     $[BC]$. Alors aire($ABC$)=$\dfrac{BC \times AK}{2}$.
   
 \end{itemize}\\
  \hline
\end{tabular}
\end{center}

\textbf{FIGURES} & \textbf{PROPRIETES}\\

  & \\ 
 & \ul{Médiatrices:}
 
 
 \begin{itemize}
     \item [$\bullet$] \ul{définition:}  La médiatrice $(d)$ d'un segment est la droite
 perpendiculaire à ce segment en son milieu. C'est un axe de symétrie du
 segment.  

\enskip

\item [$\bullet$] Un point $M$ appartient à la médiatrice d'un segment $[AB]$
si et seulement si $MA=MB$ ($M$ est équidistant de $A$ et de $B$).

\enskip

\item [$\bullet$] Les médiatrices des trois côtés d'un triangle sont
concourantes en un point $O$ qui est le centre du cercle circonscrit au
triangle (en effet: $OA=OB=OC$).

 \end{itemize}\\
 & \\
  \hline
 
 \includegraphics[width=4cm]{images/bissec1.png}
 
 
 \includegraphics[width=6cm]{images/bissec2.png} & \ul{Bissectrices:}
 
 
 \begin{itemize}
     \item [$\bullet$] \ul{définition:} La bissectrice d'un angle est la droite
     qui partage cet angle en deux angles de même mesure.
     
     \enskip
    
    \item [$\bullet$]  Un point situé sur la bissectrice d'un angle est
    équidistant des deux côtés de l'angle. On a donc pour tout point $M$ de la
    bissectrice $MQ=MR$.
    
    \enskip
    
    \item [$\bullet$] Les trois bissectrices d'un triangle sont concourantes en
    un point $I$ qui est le cercle du centre inscrit au triangle, c'est à dire
    le cercle tangent aux trois côtés. On a $IH_{A}=IH_{B}=IH_{C}$.  
 \end{itemize} \\
  & \\
  \hline
\end{tabular} 
\end{center}

 \includegraphics[width=5cm]{images/ang5.png} & \ul{Angles et cercles:} 
 $A$ et $B$ deux points d'un cercle de centre $0$:
  \begin{itemize}
    \item [$\bullet$] L'angle $\widehat{AOB}$ est un angle au centre.
    \item [$\bullet$] Si $M$ esy un point du cercle, l'angle $\widehat{AMB}$ est un

\subsection*{5) Le triangle rectangle}

\begin{center}
\begin{tabular}{|m{7cm}|m{12cm}|}
\hline
\textbf{FIGURES} & \textbf{PROPRIETES}\\
\hline

  & \\ 
 & \ul{Théorème de Pythagore:} Dans tout triangle rectangle, le carré de
 l'hypoténuse est égal à la somme des carrés des côtés de l'angle droit: Si
 $ABC$ est rectangle en $A$ alors $BC^{2}=AC^{2}+AB^{2}$.\\
  & \\
  \hline
  & \\
  & \ul{Réciproque du théorème de Pythagore:} Si le carré du plus grand côté
  d'un triangle est égal à la somme des carrés des deux autres côtés, alors ce
  triangle est rectangle et son grand côté est l'hypoténuse: Si $ABC$ est tel
  que $BC$ est le plus grand côté et $BC^{2}=AC^{2}+AB^{2}$ alors $ABC$ est
  rectangle en $A$.\\
  & \\
  \hline
 & \\
 & \ul{Trigonométrie:} Dans un triangle $ABC$ rectangle $B$ :
 
 
 \begin{itemize}
   \item [$\bullet$] \ul{définition:} $\cos \widehat{A}=\dfrac{\text{côté
   adjacent à l'angle} \widehat{A}}{\text{hypoténuse}}$
   
   
     $\cos \widehat{C}=\dfrac{\text{côté adjacent à l'angle} \widehat{C}}{\text{hypoténuse}}$  
 \end{itemize} \\
 \hline
  \end{tabular}
\end{center}
  
   \begin{center}
\begin{tabular}{|m{7cm}|m{12cm}|}
\hline
\textbf{FIGURES} & \textbf{PROPRIETES}\\
\hline
  & \\
 &   $\sin \widehat{A}=\dfrac{\text{côté
   opposé à l'angle} \widehat{A}}{\text{hypoténuse}}$
   
\enskip   
   
  $\sin \widehat{C}=\dfrac{\text{côté
   opposé à l'angle} \widehat{C}}{\text{hypoténuse}}$
   
\enskip
   
  $\tan \widehat{A}=\dfrac{\text{côté
   opposé à l'angle} \widehat{A}}{\text{côté
   adjacent à l'angle} \widehat{A}}$ 
   
\enskip
   
   $\tan \widehat{C}=\dfrac{\text{côté
   opposé à l'angle} \widehat{C}}{\text{côté
   adjacent à l'angle} \widehat{C}}$ \\
   & \\
   \hline
 &  \begin{itemize}
    \item [$\bullet$] 
    
    $\cos \widehat{A}=\sin \widehat{C}=\dfrac{AB}{AC}$ et $\cos
    \widehat{C}=\sin \widehat{A}=\dfrac{CB}{CA}$
   
   \enskip
   
    \item  [$\bullet$] $\tan \widehat{A}=\dfrac{BC}{BA}$ et $\tan
    \widehat{C}=\dfrac{BA}{BC}$
    
  
  \enskip
  
  \item  [$\bullet$] Pour tout angle aigu $\widehat{A}$ d'un triangle rectangle
  on a:
  
  
  $(\cos \widehat{A})^{2}+(\sin \widehat{A})^{2}=1$ et $\tan
  \widehat{A}=\dfrac{\sin \widehat{A}}{\cos \widehat{A}}$. 
 
\end{itemize}\\

\hline 
  & Si un triangle est inscrit dans un cercle dont le diamètre est un des côtés
  du triangle, alors ce triangle est rectangle et le diamètre est son hypoténuse.\\
  \hline
  
\end{tabular}
\end{center}

  &  \ul{Le triangle rectangle isocèle:} C'est la moitié d'un carré.
  \begin{itemize}
    \item [$\bullet$]  La diagonale d'un carré de côté $a$ a pour longueur
    $a\sqrt{2}$.
    \item [$\bullet$] $\cos (45°)=\sin (45°)=\dfrac{\sqrt{2}}{2}$.
  \end{itemize}\\
  & \\
  \hline
  & \\
  & \ul{Le triangle équilatéral:}
  \begin{itemize}
    \item [$\bullet$] La hauteur d'un triangle équilatéral de côté $a$ a pour
    longueur $\dfrac{a \sqrt{3}}{2}$.
   
   \enskip
   
   \item [$\bullet$] $\cos (60°)=\sin(30°)=\dfrac{1}{2}$
   
   \enskip
   
   \item [$\bullet$] $\cos (30°)=\sin(60°)=\dfrac{\sqrt{3}}{2}$
  \end{itemize}\\
