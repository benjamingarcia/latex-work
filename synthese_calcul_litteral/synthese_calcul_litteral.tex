\hline
\textbf{Ce que je dois savoir} & \textbf{Ce que je dois savoir faire} \\
\hline

\enskip


\begin{itemize}
  
  \item [$\bullet$] Je connais les différentes conventions d'écriture.
  
  \item[$\bullet$] Je connais la définition de ``développer''.

 \item[$\bullet$] Je connais les deux formules pour développer.


  \item[$\bullet$] Je connais la définition de ``factoriser''.
  
  \item[$\bullet$] Je connais les propriétés pour supprimer des 
  
  parenthèses.
  
  \item[$\bullet$] Je sais calculer la valeur d'une expression.
  
  \item[$\bullet$] Je sais réduire un produit.
  
  \item[$\bullet$] Je sais réduire une somme.
  
  \item[$\bullet$] Je sais reconnaître si une expression est réduite.
\end{itemize}
&

\enskip
\begin{itemize}
 
\item[$\bullet$] Je sais rétablir le signe ``$\times$'' ou le nombre ``1''.

\item[$\bullet$] Je sais reconnaître si une expression est 

développée ou
factorisée.




\item[$\bullet$] Je sais appliquer les formules pour développer (
par la méthode de mon choix).

\item[$\bullet$] Je sais trouver un facteur commun.

\item[$\bullet$] Je sais factoriser une expression.

\item[$\bullet$] Je sais supprimer des parenthèses précédées du signe ``+''.

\item[$\bullet$] Je sais supprimer des parenthèses précédées du signe ``-''.



\item[$\bullet$] Je sais réduire une expression littérale.

\item[$\bullet$] Je sais utiliser une expression littérale dans un
problème.
\end{itemize}
\\
\hline

\end{tabular}
\end{center}
