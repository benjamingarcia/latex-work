\section*{\center{Bilan Plus Grand Diviseur Commun}}




\bigskip

\begin{center}
\begin{tabular}{|m{9cm}|m{9cm}|}
\hline


\textbf{Ce que je dois savoir} & \textbf{Ce que je dois savoir faire} \\

\hline

\enskip


\begin{itemize}
  \item [$\bullet$] Je connais le vocabulaire: division euclidienne, diviseur,
  dividende, quotient et reste.
  
  
  \item [$\bullet$] Je comprends le sens des expressions: être un multiple de,
  être divisible par, être un diviseur de.
  
  \item [$\bullet$] Je connais la définition du PGCD.
  
  \item [$\bullet$] Je connais les critères de divisibilité par 2; 3; 4; 5; 9
  et 10.
  
  \item[$\bullet$] Je connais la définition de nombres premiers entre eux.
 
 
 \end{itemize}

&


\begin{itemize}
  \item[$\bullet$] Je sais effectuer une division euclidienne à la main ET à la
  calculatrice.


\item[$\bullet$] Je sais énumérer les diviseurs communs de deux entiers et
trouver leur PGCD.

\item[$\bullet$] Je sais réduire une fraction et la rendre irréductible.

\item[$\bullet$] Je sais appliquer l'algorithme des soustractions successives.

\item[$\bullet$] Je sais appliquer l'algorithme d'Euclide.

  \item[$\bullet$] Je sais résoudre des problèmes.
 
\end{itemize} \\

\hline

\end{tabular}
\end{center}
