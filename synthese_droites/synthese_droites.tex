\center{\textbf{\Large{Bilan droites}}}




\bigskip

\begin{center}
\begin{tabular}{|m{9cm}|m{10cm}|}
\hline
\textbf{Ce que je dois savoir} & \textbf{Ce que je dois savoir faire} \\
\hline

\enskip


\begin{itemize}
  \item[$\bullet$] Je connais les notations mathématiques pour les droites,
  demi-droites et segments.
  \item [$\bullet$] Je connais les propriétés sur le nombre de droites passant
  par un ou deux points.
  \item [$\bullet$] Je connais la définition de points alignés.
  \item [$\bullet$] Je connais la définition de points d'intersection et de
  droites sécantes.
  \item [$\bullet$] Je connais la définition de droites perpendiculaires et
  droites parallèles.
  \item [$\bullet$] Je connais les trois propriétés sur les droites parallèles
  et perpendiculaires.

\end{itemize}
&

\enskip
\begin{itemize}
  \item[$\bullet$] Je sais reconnaître une droite, une demi-droite et un
  segment.
  \item[$\bullet$] Je sais utiliser le symbole ``$\in$''.
  \item[$\bullet$] Je sais reconnaître si un point appartient à une droite ou à
  une demi-droite ou à un segment.
 \item[$\bullet$] Je sais coder des droites parallèles et des droites
 perpendiculaires.
  \item[$\bullet$] Je sais tracer la perpendiculaire à une droite passant par
  un point.
   \item[$\bullet$] Je sais tracer la parallèle à une droite passant par
  un point.
  \item[$\bullet$] Je sais réaliser une figure à partir d'un schéma codé.
  \item[$\bullet$] Je sais écrire un programme de construction.
  \item[$\bullet$] Je sais résoudre des problèmes.
    \end{itemize} \\
\hline 

\end{tabular}
\end{center}
