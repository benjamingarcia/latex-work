\subsection*{Exercice 4}


\begin{enumerate}
  \item $\mathcal{C}1$ est le cercle de centre $B$ et de rayon $5$. Le point
  $C$ est sur $\mathcal{C}1$, donc $BC=5$.
  \item Dans le cercle $\mathcal{C}2$, on a: $C$ est un point de $\mathcal{C}2$
  et $OB$ est un diamètre de $\mathcal{C}2$ donc le triangle $OBC$ est un
  triangle rectangle en $C$.
  \item \begin{enumerate}
          \item Dans le triangle $OBC$ rectangle en $C$, nous utilisons le
          théorème de Pythagore:
         \begin{equation*}
          OB^{2} = BC^{2}+OC^{2}
          \end{equation*}
\begin{equation*}
          6^{2} = 5^{2}+OC^{2}
          \end{equation*}
          \begin{equation*}
          OC^{2} = 6^{2}-5^{2}=36-25=11
        \end{equation*}
Donc $OC= \sqrt{11}$.
\item $\mathcal{C}3$ est le cercle de centre $O$ et de rayon $OC$. Le point $D$
est sur $\mathcal{C}3$ donc $OD=OC=\sqrt{11}$. $x$ étant l'abscisse du point
$D$, on trouve $x=\sqrt{11}$.
\item $x$ est un irrationnel.
\end{enumerate}
\item D'après les prorpiétés des médiatrices, $A$ est le milieu de $[OI]$. Or
$O$ est le point d'abscisse $0$, $I$ est le point d'abscisse $1$. On en déduit
que l'abscisse de $A$ est $\dfrac{1}{2}=0,5$ qui est un nombre décimal.

\end{enumerate}
