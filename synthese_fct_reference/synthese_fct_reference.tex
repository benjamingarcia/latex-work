\section*{\center{Bilan fonctions de référence}}




\bigskip
\begin{center}
\begin{tabular}{|m{9cm}|m{10cm}|}
\hline
\textbf{Ce que je dois savoir} & \textbf{Ce que je dois savoir faire} \\
\hline

\enskip


\begin{itemize}
  \item[$\bullet$] Je connais le vocabulaire
  \item [$\bullet$] Je sais représenter graphiquement la fonction
  carré et de la fonction inverse
  \item [$\bullet$] Je connais les propriétés des fonctions carré et inverse
  (sens de variation, parité, domaine de définition)
 
\end{itemize}
&

\enskip
\begin{itemize}
  \item[$\bullet$] Je sais reconnaître la courbe représentative de la fonction
  carré
  \item[$\bullet$] Je sais reconnaître la courbe représentative de la fonction
  inverse
  \item[$\bullet$] Je sais déduire le sens de variation de certaines fonctions
  à partir de celles de la fonction carré et de la fonction inverse
 \item[$\bullet$] Je sais résoudre des équations du type $(x-2)^2=16$ et les
 inéquations associées (avec un tableau de signes)
  
 \end{itemize} \\
\hline

\end{tabular}
\end{center}
