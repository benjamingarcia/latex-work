\begin{tabular}{|m{95mm}|m{95mm}|}
\hline
\textbf{Ce que je dois savoir} & \textbf{Ce que je dois savoir faire} \\
\hline

\enskip


\begin{itemize}
  \item[$\bullet$] Je connais les notations pour un segment: 
  
  segment et
  longueur.
  \item [$\bullet$] Je connais la définition de milieu d'un segment.
  \item [$\bullet$] Je connais le vocabulaire du cercle: centre, rayon,
  diamètre, arc de cercle et corde.
  \item [$\bullet$] Je sais calculer un rayon connaissant le diamètre.
  \item [$\bullet$] Je sais calculer un diamètre connaissant le rayon.
  \item [$\bullet$] Je connais les propriétés du cercle.
  \item [$\bullet$] Je connais le vocabulaire: équidistant, médiatrice.
  \item [$\bullet$] Je connais les 2 propriétés de la médiatrice.
  \item [$\bullet$] Je sais coder une figure.

\end{itemize}
&

\enskip
\begin{itemize}
  \item[$\bullet$] Je sais reconnaître deux longueurs identiques à partir du
  codage.
  \item[$\bullet$] Je sais calculer des longueurs de segments.
  \item[$\bullet$] Je sais placer des points à une distance donnée en utilisant
  le compas.
 \item[$\bullet$] Je sais construire un triangle connaissant les trois
 longueurs.
  \item[$\bullet$] Je sais construire la médiatrice d'un segment (avec
  l'équerre et avec le compas).
   \item[$\bullet$] Je sais utiliser les propriétés des médiatrices pour
   résoudre des problèmes.
  \item[$\bullet$] Je sais réaliser une figure à partir d'un programme de
  construction.
  \item[$\bullet$] Je sais écrire un programme de construction.
    \end{itemize} \\
\hline 

\end{tabular}
\end{center}
