\section{Produit d'un nombre négatif par un nombre positif}

On considère l'expression $B=(-2)+(-2)+(-2)+(-2)$.

\begin{enumerate}
  \item Calculer B.
  \item  Ecrire B sous la forme d'un produit.
  \item Ecrire les expressions suivantes sous la forme d'une somme et
  les calculer:
  
  \begin{center}
  $C=(-6) \times 3$ \qquad \qquad $D=(-22) \times 5$ \qquad \qquad $E=(-7)
  \times 7$ \qquad \qquad $F=(-1,5) \times 6$
  \end{center}
  
  \item Conjecturer la manière dont on calcule le produit d'un nombre négatif
  par un nombre positif.
\end{enumerate}


\section{Conjecture sur le produit}

\begin{enumerate}
  \item Sur le tableau:
  
  \begin{enumerate}

    (en haut à droite).
    \item En utilisant le résultat de la première partie, compléter la partie
    qui concerne le produit d'un nombre négatif par un nombre positif (en bas à
    droite).
    \item Compléter entièrement le tableau (expliquez vos
    choix).
\end{enumerate}


\item Application sur quelques exemples:

\begin{enumerate}
 \item A l'aide de la table, donner le résultat des calculs suivants:
 
 \begin{center} 
 $A=(-5) \times 4$ \qquad \qquad $B=3 \times (-2)$ \qquad \qquad $C=5 \times
 (-4)$ \qquad \qquad $D=(-1) \times (-3)$
 \end{center}
  
 \item En s'inspirant de ce qui précède, proposer un résultat pour les calculs
 suivants: 
 
 \begin{center}
 $E=(-9,2) \times 2$ \qquad \qquad $F=1,5 \times (-8)$ \qquad \qquad $G=(-3,14)
 \times 0$ \qquad \qquad $H=(-1,2) \times (-0,1)$
 \end{center}
 
 \item Vérifier les résultats à la calculatrice.
  \end{enumerate}
  
  \item Proposer une règle qui permet, dans tous les cas, de calculer le produit
  de deux nombres relatifs.
\end{enumerate}
