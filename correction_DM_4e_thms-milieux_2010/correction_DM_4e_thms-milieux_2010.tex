\subsection*{Exercice 2}


\begin{enumerate}
  \item  \ul{\textit{Données:}}Dans le triangle ABC, on sait que
  J est le milieu de [AB], 
    I est le milieu de [AC] et BC=6cm. 
    
 
    
 \ul{\textit{Propriété:}} Si dans un
    triangle, un segment joint les milieux de deux côtés alors sa longueur est égale à la moitié de celle 
 du troisième côté.
 
 \ul{\textit{Conclusion:}}  
   $IJ=\dfrac{BC}{2}=\dfrac{6}{2}=3cm$.
 
  \enskip

  \item   J est le milieu de [AB] et AB=8cm donc BJ=4cm.
  
\enskip
  
  \ul{\textit{Données:}} Dans le triangle BCJ, on sait que
   K est le milieu de [BC], 
    L est le milieu de [JC] et BJ=4cm. 
    
    
    
    
    
   \ul{\textit{Propriété:}} Si dans un
    triangle, un segment joint les milieux de deux côtés alors sa longueur est égale à la moitié de celle 
 du troisième côté.
 
  \ul{\textit{Conclusion:}}  $LK=\dfrac{BJ}{2}=\dfrac{4}{2}=2cm$.  
\end{enumerate}


\subsection*{Exercice 3}

\begin{enumerate}
  \item  \ul{\textit{Données:}} Dans le triangle ABC, on sait que P est le
  milieu de [AB]  et R est le milieu de [AC]. 

\ul{\textit{Propriété:}} Si dans un triangle, une droite passe par
les milieux de deux côtés alors elle est parallèle au troisième côté.

\ul{\textit{Conclusion:}}  Les droites (PR) et (BC) sont parallèles.  

\enskip

\item  Les points B, L et C sont alignés donc les droites (BL) et (BC)
  sont confondues (c'est-à-dire (BL)=(BC)). Commme (BC) et (PR) sont parallèles
  (d'après la question 1), les droites (BL) et (PR) sont parallèles.
  
  \enskip
  
  
  \item  \ul{\textit{Données:}}Dans le triangle KPR, on sait que les droites
  (BL) et (PR) sont parallèles et que B est le milieu de [PK].  

\ul{\textit{Propriété:}} Si,
  dans un triangle, une droite passe par le milieu  d'un côté et est parallèle à un second côté alors elle passe
  par le milieu du troisième côté.
  
   \ul{\textit{conclusion:}} L est le milieu de [KR].
  
  \enskip
  
  \item \ul{\textit{Données:}} Dans le triangle ABC, on sait que P est le milieu
  de [AB] et R le milieu de [AC]. 

\ul{\textit{Propriété:}} Si, dans un triangle, un segment joint le
   milieu de deux côtés alors sa longueur est égale à la moitié de celle du troisième
   côté.
   
    
   
 \ul{\textit{conclusion:}} $PR=\dfrac{BC}{2}=\dfrac{18}{2}=9cm$.
   
   \enskip
   
   
 \item \ul{\textit{Données:}} Dans le triangle KPR, on sait que B est le milieu
 de [KP] et L le milieu de [KR]. 
  
 
  \ul{\textit{Propriété:}} Si, dans un triangle, un segment joint le
   milieu de deux côtés alors sa longueur est égale à la moitié de celle du troisième
   côté.
   
  \ul{\textit{conclusion:}}  $BL=\dfrac{PR}{2}=\dfrac{9}{2}=4,5cm$.
\end{enumerate}

\begin{enumerate}
\item [2.(b)] \ul{\textit{Données:}} Dans le triangle GAZ, on sait que F est
  le milieu de [AZ]  et E est le milieu de [GZ]. 

\ul{\textit{Propriété:}} Si dans un triangle, une droite passe par
les milieux de deux côtés alors elle est parallèle au troisième côté.

\ul{\textit{Conclusion:}}  Les droites (EF) et (AG) sont parallèles.

\enskip

\item [2.(c)] Les droites (ER)et (AZ) sont parallèles et les points A, F et Z
sont alignés donc les droites (ER) et (AF) sont aussi parallèles. De même, on
trouve que (EF) et (AR) sont parallèles.

\enskip

\ul{\textit{Données:}} FERA est un quadrilatère vérifiant: (RE) // (AF) et (RA)
// (EF)



\ul{\textit{Propriété:}} Si un
quadrilatère a ses côtés opposés parallèles alors c'est un parallélogramme.

\ul{\textit{Conclusion:}}  FERA est un parallélogramme.

\enskip

\ul{\textit{Données:}} FERA est un parallélogramme et $\widehat{RAF}$ est un
angle droit.



\ul{\textit{Propriété:}} Si un
parallélogramme a un angle droit alors c'est un rectangle.

\ul{\textit{Conclusion:}}  FERA est un rectangle. 
\end{enumerate}
