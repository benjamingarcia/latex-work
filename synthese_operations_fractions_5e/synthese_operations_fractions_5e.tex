\bigskip

\begin{center}
\begin{tabular}{|m{9cm}|m{9cm}|}
\hline


\textbf{Ce que je dois savoir} & \textbf{Ce que je dois savoir faire} \\

\hline

\enskip


\begin{itemize}
  \item [$\bullet$] Je connais la règle pour additionner et soustraire deux
  nombres en écriture fractionnaire.
  
  \item [$\bullet$] Je sais réduire au même dénominateur.
  
  \item [$\bullet$] Je sais simplifier une fraction.
  
  \item [$\bullet$] Je connais la méthode pour multiplier deux nombres en
  écriture fractionnaire.
 
 \item [$\bullet$] Je sais écrire un nombre entier sous forme de fraction.

 \end{itemize}

&

  \enskip

\begin{itemize}
  
  

   
  
  \item[$\bullet$] Je sais additionner et soustraire des
  nombres en écriture fractionnaire de même dénominateur.


  \item[$\bullet$] Je sais additionner et soustraire des
  nombres en écriture fractionnaire ayant des  dénominateurs 
  différents.

  
      
  \item[$\bullet$] Je sais multiplier des
  nombres en écriture fractionnaire.
  
  \item[$\bullet$] Je sais simplifier un produit ``au fur et à mesure''.
  
  \item[$\bullet$] Je sais effectuer un calcul en respectant les règles
  de priorités.
 
\end{itemize} \\

\end{tabular}
