\subsection*{Exercice}

\begin{enumerate}
  \item On considère la fonction $f$ définie sur $\mathbb{R}$ par:
  $f(x)=(x-1)^2-2$.
  \begin{enumerate}
    \item Compléter et expliquer chaque étape du raisonnement suivant:
    

    Soit  $a< b<1$  
    
    
    donc  $a-1$ \ldots \ldots   $b-1$ \ldots \ldots,    \qquad    explication:
    
    donc  $(a-1)^2$  \ldots \ldots $(b-1)^2$,  \qquad  \qquad explication:
    
    donc $(a-1)^2-2$  \ldots \ldots $(b-1)^2-2$,  \quad explication:
    
    \item Quelle propriété de $f$ a-t-on démontrée?
  \end{enumerate}
  
  \item On considère la fonction $g(x)=3+\dfrac{1}{x-2}$ définie sur
  $\mathbb{R}$ \textbackslash \{2\}
  
  \begin{enumerate}
    \item Ecrire ou compléter puis expliquer chacune des étapes:
    
    Soit $2<a<b$
    
    
    donc:
    
    \bigskip
    
    \bigskip
    
    
    \bigskip
    
    
    donc: $3+ \dfrac{1}{a-2} \ldots \ldots 3+ \dfrac{1}{b-2}$
    \item Quelle propriété de $g$ a-t-on démontrée?
  \end{enumerate}
  \item Reprendre le même type de raisonnement pour dresser le tableau de
  variation des fonctions: 
  
  $h(x)=(x+2)^2-1$ et $i(x)=2+\dfrac{1}{x-1}$.
  \end{enumerate}
