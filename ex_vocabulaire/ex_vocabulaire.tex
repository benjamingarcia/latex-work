\ul{Exercice 13:} Soit $A=4n$ et $B=4+n$.

\begin{enumerate}
  \item Réécrire l'expression $A$ en faisant apparaître le signe ``$\times$''
  qui a été supprimé.
  \item Calculer $A$ et $B$ pour les valeurs suivantes de $n$:
  
  \begin{enumerate}
    \item $n=8$
    \item $n=11$
    \item $n=0$
    \item $n=1,2$
\end{enumerate}

\end{enumerate}
