 \textbf{\Large{\ul{Proportionnalité}}}
\end{center}

\enskip

\textbf{\large{1. Définitions}}

\enskip

\ul{Définition}: Dire que deux grandeurs sont proportionnelles revient à dire
que les valeurs de l'une s'obtiennent en multipliant les valeurs de l'autre
par un même nombre appelé \textbf{coefficient de proportionnalité}.

\enskip


\ul{Définition}: Un \textbf{tableau de proportionnalité} est un tableau
représentant deux grandeurs proportionnelles.

\enskip

\ul{Exemple}: 

\enskip


\begin{tabular}{|c|c|c|c|c|c|}
\hline
côté du carré en cm & \quad 1 \quad & \quad 2 \quad & \quad 3 \quad & \quad 6
\quad & \quad 10 \quad \\
\hline
périmètre en cm & \quad & \quad & \quad & \quad &\quad \\
\hline
\end{tabular}

\bigskip

\textbf{\large{2. Calculer une quatrième proportionnelle}}


\enskip

\ul{Exemple}: Le prix du pain est proportionnel au nombre de baguettes achetées.

\bigskip


\textbf{Méthode 1: En utilisant le coefficient de proportionnalité}

\enskip


\begin{tabular}{|c|c|c|c|}
\hline
nombre de baguettes & \quad 1 \quad & \quad 3 \quad  & \quad 4 \quad \\
\hline
prix en \euro & \quad 0,87 \quad & \quad 2,61 \quad  & \quad \ldots \ldots \quad \\
\hline 
\end{tabular}


\bigskip

\textbf{Méthode 2: En utilisant les opérations dans un tableau de
proportionnalité}

\enskip

\ul{Propriété}: Dans un tableau de proportionnalité, on peut construire une
nouvelle colonne:

\begin{itemize}
  \item [$\bullet$] en \textbf{multipliant} les valeurs d'une colonne par un
  même nombre;
  \item [$\bullet$] en \textbf{divisant} les valeurs d'une colonne par un
  même nombre; 
  \item [$\bullet$] en \textbf{ajoutant} les valeurs de deux colonnes ensemble. 
\end{itemize}

\bigskip



\begin{tabular}{cc}
\begin{minipage}{8cm}
\begin{tabular}{|c|c|c|}
\hline
nombre de baguettes  & \quad 4 \quad & \quad 8 \quad   \\
\hline
prix en \euro & \quad 3,48 \quad   & \quad \ldots \ldots \quad \\
\hline 
\end{tabular}


\end{minipage}
&
\begin{minipage}{10cm}
\begin{tabular}{|c|c|c|c|}
\hline
nombre de baguettes  & \quad 3 \quad & \quad 5 \quad  & \quad 8 \quad \\
\hline
prix en \euro & \quad 2,61 \quad & \quad 4,35 \quad  & \quad \ldots \ldots \quad
\\
\hline 
\end{tabular}
\end{minipage}
\end{tabular}

\bigskip

\textbf{Méthode 3: En utilisant la règle ``d'égalité des produits en

\enskip
