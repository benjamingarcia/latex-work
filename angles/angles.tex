\section{Vocabulaire}

\subsection{Angles complémentaires, angles supplémentaires}


\ul{Définition}: Deux angles sont \textbf{complémentaires} lorsque la somme de
leurs mesures est 90°.

\enskip

\ul{Définition}: Deux angles sont \textbf{supplémentaires} lorsque la somme de
leurs mesures est 180°.



\subsection{Angles adjacents}

\begin{tabular}{cc}
\begin{minipage}{11cm}
\ul{Définition}: Deux angles \textbf{adjacents} ont un sommet commun, un
côté commun et  sont situés de part et d'autre de ce côté.
 \end{minipage}&
\begin{minipage}{7cm}

\end{minipage}
\end{tabular}


\subsection{Angles opposés par le sommet}

\begin{tabular}{cc}
\begin{minipage}{12cm}
\ul{Définition}: Deux angles \textbf{opposés par le sommet} ont le même sommet
et des côtés dans le prolongement l'un de l'autre.

\medskip

\ul{Propriété}: Si deux angles sont opposés par le sommet alors ils ont la même
mesure.
 \end{minipage}&
\begin{minipage}{6cm}

\end{minipage}
\end{tabular}


\subsection{Angles alternes-internes, angles correspondants}

\begin{tabular}{cc}
\begin{minipage}{12cm}
\ul{Définition}: La sécante $(\Delta)$ coupe les droites $(d)$ et $(d')$ en A et
B.

 Deux angles \textbf{alternes-internes} sont:
 
 $\bullet$ entre $(d)$ et $(d')$ 
 
 $\bullet$ de part
 et d'autre de la sécante $(\Delta)$.
 
 
 \medskip
 
 \ul{Définition}: La sécante $(\Delta)$ coupe les droites $(d)$ et $(d')$ en A et
B.

 Deux angles sont \textbf{correspondants} si:
 
 $\bullet$ ils sont du même côté de la sécante  $(\Delta)$
 
 $\bullet$ un angle est situé entre $(d)$ et $(d')$ et l'autre non.
 \end{minipage}&
\begin{minipage}{6cm}

\end{minipage}
\end{tabular}



\section{Parallèles, sécantes et angles}

\subsection{Reconnaître des angles de même mesure}

\ul{Propriété}: Si deux droites \textbf{parallèles} sont coupées par une
sécante, alors les \textbf{angles alternes-internes} qu'elles forment ont
\textbf{même mesure}.

\enskip

\ul{Exemple}:

\enskip

\begin{tabular}{cc}
\begin{minipage}{9cm}
\textit{Données:}

\bigskip

\bigskip

\bigskip

$\bullet$ $(d_1)$ et $(d_2)$ sont parallèles

$\bullet$ $\widehat{a}$ et $\widehat{b}$ sont des angles alternes-internes
\end{minipage}
&
\begin{minipage}{9cm}
 \qquad \qquad \textit{Conclusion:}


\bigskip

\bigskip

\bigskip


\qquad \qquad $\widehat{a}=\widehat{b}$
\end{minipage}
\end{tabular}





\ul{Propriété}: Si deux droites \textbf{parallèles} sont coupées par une
sécante, alors les \textbf{angles correspondants} qu'elles forment ont
\textbf{même mesure}.


\enskip

\ul{Exemple}:

\enskip

\begin{tabular}{cc}
\begin{minipage}{9cm}
\textit{Données:}

\bigskip

\bigskip

\bigskip

$\bullet$ $(d_1)$ et $(d_2)$ sont parallèles

$\bullet$ $\widehat{a}$ et $\widehat{b}$ sont des angles correspondants
\end{minipage}
&
\begin{minipage}{9cm}
 \qquad \qquad \textit{Conclusion:}


\bigskip

\bigskip

\bigskip

\qquad \qquad $\widehat{a}=\widehat{b}$
\end{minipage}
\end{tabular}

\subsection{Reconnaître des droites parallèles}

\ul{Propriété}: Si deux droites coupées par une sécante forment deux
\textbf{angles alternes-internes de même mesure} alors ces droites sont
\textbf{parallèles}.


\enskip

\ul{Exemple}:

\enskip

\begin{tabular}{cc}
\begin{minipage}{9cm}
\textit{Données:}

\bigskip

\bigskip

\bigskip

$\bullet$ $\widehat{a}$ et $\widehat{b}$ sont des angles alternes-internes

$\bullet$ $\widehat{a}=\widehat{b}$
\end{minipage}
&
\begin{minipage}{9cm}
 \qquad \qquad \textit{Conclusion:}


\bigskip

\bigskip

\bigskip

\qquad \qquad $(d_1)$ et $(d_2)$ sont parallèles. 
\end{minipage}
\end{tabular}

\bigskip

\bigskip

\ul{Propriété}: Si deux droites coupées par une sécante forment deux
\textbf{angles correspondants de même mesure} alors ces droites sont
\textbf{parallèles}.


\enskip

\ul{Exemple}:

\enskip

\begin{tabular}{cc}
\begin{minipage}{9cm}
\textit{Données:}

\bigskip

\bigskip

\bigskip


$\bullet$ $\widehat{a}$ et $\widehat{b}$ sont des angles correspondants

$\bullet$ $\widehat{a}=\widehat{b}$
\end{minipage}
&
\begin{minipage}{9cm}
 \qquad \qquad \textit{Conclusion:}


\bigskip

\bigskip

\bigskip


\qquad \qquad $(d_1)$ et $(d_2)$ sont parallèles. 
\end{minipage}
\end{tabular}

\bigskip

\bigskip

\ul{Remarque}: On dit que ces propriétés sont les réciproques des propriétés
énoncées au paragraphe précédent (reconnaître des angles de même mesure). Les
données et la conclusion d'une propriété et de sa réciproque sont échangées.
