\hline
\textbf{Ce que je dois savoir} & \textbf{Ce que je dois savoir faire} \\
\hline

\enskip


\begin{itemize}
  \item[$\bullet$] Je connais le vocabulaire: dizaine, unité, dixième,
  centième \ldots
  \item [$\bullet$] Je sais écrire un nombre en lettres.
  \item [$\bullet$] Je sais trouver la partie entière et la partie 
  
  décimale
  d'un nombre.
  \item [$\bullet$] Je sais écrire une fraction décimale en nombre décimal.
  \item [$\bullet$] Je connais la règle des zéros inutiles.
  \item [$\bullet$] Je sais lire l'abscisse d'un point sur une droite graduée.
  \item [$\bullet$] Je connais les méthodes pour comparer deux nombres.
\end{itemize}
&

\enskip
\begin{itemize}
  \item[$\bullet$] Je sais placer un nombre dans le tableau (dizaine, unité,
  dixième\ldots)
  \item[$\bullet$] Je sais reconnnaître le chiffre des unités, des dizaines,
  des dixièmes\ldots
  \item[$\bullet$] Je sais décomposer un nombre décimal.
 \item[$\bullet$] Je sais écrire un nombre sous forme de fractions.
  \item[$\bullet$] Je sais placer un point sur une droite graduée.
   \item[$\bullet$] Je sais comparer des nombres, les ranger dans l'ordre
   croissant ou décroissant.
  \item[$\bullet$] Je sais encadrer un nombre.
    \end{itemize} \\
\hline 

\end{tabular}
