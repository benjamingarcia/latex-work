\begin{center} 
\textbf{Signes de quotients}
\end{center}

\begin{enumerate}
  \item \begin{enumerate}
          \item Calculer les quotients: $(-3) \div 4$; \quad $3 \div (-4)$;
          \quad $-(3 \div 4)$.
          \item Que peut-on en déduire pour les nombres $\dfrac{-3}{4}$; \quad
          $\dfrac{3}{-4}$ \quad  et \quad $-\dfrac{3}{4}$ ?
\end{enumerate}
\item De la même façon, que peut-on dire des nombres $\dfrac{-72}{-16}$;
\quad $\dfrac{72}{16}$;  \quad $-\dfrac{-72}{16}$ \quad et
\quad $-\dfrac{72}{-16}$ ?
 
 Justifier votre réponse.
\end{enumerate}
