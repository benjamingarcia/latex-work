\section*{\center{Bilan fonction linéaire}}




\bigskip

\begin{center}
\begin{tabular}{|m{9cm}|m{9cm}|}
\hline


\textbf{Ce que je dois savoir} & \textbf{Ce que je dois savoir faire} \\

\hline

\enskip


\begin{itemize}
  
  
  \item [$\bullet$] Je reconnais un tableau de proportionnalité.
  
  \item [$\bullet$] Je sais effectuer des opérations sur les colonnes d'un
  tableau de proportionnalité.

\item [$\bullet$] Je sais reconnaître graphiquement une situation de
proportionnalité.

\item [$\bullet$] Je sais reconnaitre une fonction linéaire à partir d'une
expression algébrique.

\item [$\bullet$] Je sais le lien entre tableau de proportionnalité et
tableau de valeurs d'un fonction linéaire.

\item [$\bullet$] Je connais le vocabulaire: coefficient linéaire, coefficient
de proportionnalité et coefficient directeur.

\item [$\bullet$] Je sais reconnaitre graphiquement une fonction linéaire.

\item [$\bullet$]	Je sais tracer la représentation graphique d'une fonction
linéaire.

\item [$\bullet$]	Je connais les formules d'augmentation et baisse en
pourcentage.
 \end{itemize}

&


\begin{itemize}
  
  
  \item [$\bullet$] Je sais trouver un coefficent de proportionnalité.
  
  \item[$\bullet$] Je sais déterminer une $4^e$ proportionnelle.


\item[$\bullet$] Je sais résoudre des problèmes: agrandissement, réduction,
échelle, vitesse moyenne et grandeurs composées.

  \item[$\bullet$] Je sais trouver l'expression algébrique d'une fonction
  linéaire à partir d'une égalité (ou d'un point de la courbe).

\item[$\bullet$] Je sais trouver l'expression algébrique d'une fonction
  linéaire à partir d'un tableau de valeurs.
 
\item[$\bullet$] Je sais trouver l'expression algébrique d'une fonction
  linéaire à partir de sa représentation graphique.

  \item[$\bullet$] Je sais calculer image et antécédent d'une fonction linéaire.
 
  
      
  \item[$\bullet$] Je sais appliquer un pourcentage.
  
  \item[$\bullet$] Je sais calculer un pourcentage.
  
  \item[$\bullet$] Je sais résoudre des problèmes de pourcentage.
 
\end{itemize} \\

\hline

\end{tabular}
\end{center}
