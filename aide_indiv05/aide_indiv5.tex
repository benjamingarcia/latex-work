\section*{\center{Aide individualisée: développer et factoriser}}

\subsection*{Rappels}


\begin{itemize}
  \item [$\bullet$] \ul{Réduire une somme} : c'est écrire cette somme avec le
  moins de termes possibles , en regroupant les termes de même nature. Exemple:
  $3x^{2}+7x+4-5x=$
  
  \enskip
  
  \item [$\bullet$] \ul{Développer un produit} : c'est écrire ce produit sous
  la forme d'une somme . 
  
  
  Exemple: $(2x+1)(x-5)=$
  
  \enskip
  
  \item [$\bullet$] \ul{Factoriser une somme} : c'est écrire cette somme sous la
  forme d'un produit. 
  
  
  Exemple: $(a+2)(3a-5)+3(a+2)=$
  \end{itemize}

\begin{itemize}
  \item [$\bullet$] Ne pas oublier le \textbf{double produit} dans le
  développement d'un carré
  \item [$\bullet$] Un signe '-' devant une parenthèse: il faut changer tous
  les signes. 
  
  Exemple: $-(x-7+2x^{2})=-x+7-2x^{2}$
\end{itemize}
\subsection*{Applications}

\textbf{Exercice 1:} Développer,  réduire et ordonner chacune des expressions
suivantes:


$A=(x-1)(-x+2)+(2x+1)^{2}$

$B=(x-3)^{2}-4x(x-1)$

$C=(2x+5)^{2}-(5x+2)(5x-2)-(1-x)(3+x)$

$D=(3x-1)^{2}$

\bigskip

\textbf{Exercice 2:} 
\begin{enumerate}
  \item Reconnaître les développements de carré et en donner la factorisation:
  
  
  $E=x^{2}-4x+4$  \qquad \qquad $F=6x+9+x^{2}$ \qquad \qquad
  $G=\dfrac{x^{2}}{4}-3x+9$
  \item Reconnaître la différence entre deux carrés et en donner la
  factorisation:
  
  $H=25x^{2}-4$ \qquad \qquad $I=-4x^{2}+9$ \qquad \qquad $J=3-x^{2}$
\end{enumerate}
