\section{Vecteurs du plan}

\subsection{Vecteurs }

\textit{Ex 1}

\enskip

\ul{Définitions}:
\begin{itemize}
  \item [$\bullet$]  Si $A$ et $B$ sont deux points distincts, le vecteur
  $\overrightarrow{AB}$ est caractérisé par:
  \begin{itemize}
    \item sa \textbf{direction}: celle de $(AB)$ et de toutes les droites
    parallèles à $(AB)$
    \item son \textbf{sens}: celui de $A$ vers $B$
    \item sa \textbf{longueur}: la distance $AB$.
  \end{itemize}
  
 \item[$\bullet$] Si $A=B$, le vecteur  $\overrightarrow{AA}$ s'appelle
 \textbf{vecteur nul} et est noté $\vec{0}$.
 
 Le vecteur nul n'a ni direction, ni sens. Sa longueur est égale à 0.
  
\end{itemize}

\enskip

\ul{Remarque}: Au lieu de parler de la longueur d'un vecteur, on préfère parler
de sa norme. 

La norme de $\overrightarrow{AB}$ se note
$||\overrightarrow{AB}||$. On a: $AB=||\overrightarrow{AB}||$.


\subsection{Egalité de deux vecteurs}

\ul{Définition}:

\begin{tabular}{cc}
\begin{minipage}{12cm}
On dit que deux vecteurs $\overrightarrow{AB}$ et $\overrightarrow{AB}$ sont
égaux dans chacun des cas suivants:
\begin{itemize}
  \item [$\bullet$] Ils ont même direction, même sens et même longueur.
  \item [$\bullet$] $ABCD$ est un parallélogramme (éventuellement aplati).
  \item [$\bullet$] C'est la même translation qui transforme $A$ en $B$ et $C$
  en $D$.
\end{itemize}

On note alors: $\overrightarrow{AB}$=$\overrightarrow{CD}=\vec{u}$
\end{minipage}
&
\begin{minipage}{6cm}
\includegraphics[width=4cm]{images/ex12.png}
\end{minipage}
\end{tabular}

\enskip


\ul{Remarque}: Un vecteur n'est donc en aucun cas lié à un point origine ou
extrémité. Il y a donc une infinité de représentants de $\vec{u}$.

\enskip
  
\textit{Ex 2}


\subsection{Addition de deux vecteurs}

Soit $\vec{u}$ et $\vec{v}$ deux vecteurs quelconques, $A$ un point du plan, $B$
et $C$ les points définis par $\overrightarrow{AB}=\vec{u}$ et
$\overrightarrow{BC}=\vec{v}$. On appelle somme de $\vec{u}$ et $\vec{v}$,
notée $\vec{u}+\vec{v}$, le vecteur qui peut-être représenté par l'une des deux
règles suivantes:

\enskip

\ul{Relation de Chasles}: L'égalité
$\overrightarrow{AB}+\overrightarrow{BC}=\overrightarrow{AC}$ s'appelle la \textbf{relation de Chasles}. $\vec{u}+\vec{v}=\overrightarrow{AC}$.

Les vecteurs $\overrightarrow{AB}$ et $\overrightarrow{BC}$ sont ``mis bout à
bout''.

\begin{center}
\includegraphics[width=18cm]{images/chasles.jpg}
\end{center}
 
\enskip

\ul{Règle du parallélogramme}: $\vec{u}+\vec{v}$ représente la diagonale du
parallélogramme;
$\vec{u}+\vec{v}=\overrightarrow{AB}+\overrightarrow{BC}=\overrightarrow{AD}$
où $D$ est tel que $ABCD$ est un parallélogramme (éventuellement aplati).

\begin{center}
\includegraphics[width=18cm]{images/regle_pgm.jpg}
\end{center}


\enskip

\ul{Remarques}:

\begin{tabular}{cc}
\begin{minipage}{115mm}
\begin{itemize}
  \item [$\bullet$] $\vec{u}+\vec{v}=\vec{v}+\vec{u}$ et 
  $(\vec{u}+\vec{v})+\vec{w}=\vec{u}+(\vec{v}+\vec{w})$
  \item [$\bullet$] Si $\vec{u}=\overrightarrow{AB}$, l'opposé est le vecteur
  $\overrightarrow{BA}$ noté $-\vec{u}$. 
  
  On a
  $-\vec{u}=\overrightarrow{BA}=-\overrightarrow{AB}=\overrightarrow{AB'}$.
  \item [$\bullet$] $\vec{u}+(-\vec{u})=\vec{0}$
  \item [$\bullet$] La différence de deux vecteurs est la somme du premier avec
  l'opposé du second: $\vec{u}-\vec{v}=\vec{u}+(-\vec{v})$
  \item [$\bullet$] La relation de Chasles sera utile pour décomposer un
  vecteur: pour tout point $M$, on a
  $\overrightarrow{AC}=\overrightarrow{AM}+\overrightarrow{MC}$
\end{itemize}
\end{minipage}

\begin{minipage}{7cm}
\includegraphics[width=7cm]{images/ex13.png}
\end{minipage}
\end{tabular}

\section{Multiplication par un réel, colinéarité}

\subsection{Multiplication par un réel}

\ul{Exemple}: $\overrightarrow{AB}=\vec{u}$;
$\overrightarrow{AC}=\vec{u}+\vec{u}+\vec{u}=3\vec{u}$ et
$\overrightarrow{CD}=2\vec{u}$; $\overrightarrow{DA}=-5\vec{u}$.

\includegraphics[width=7cm]{images/ex21.png}




\enskip

\ul{Définition}: Le produit du vecteur $\vec{u}$ par le nombre réel $k$ est le
\textbf{vecteur} noté $k\vec{u}$ tel que:

\begin{tabular}{cc}
\begin{minipage}{12cm}
\begin{itemize}
  \item [$\bullet$] si $k>0$ et $\vec{u} \neq \vec{0}$, $k\vec{u}$ a la même
  direction et le même sens que $\vec{u}$ et a pour norme $k||\vec{u}||$
  \item [$\bullet$] si $k<0$ et $\vec{u} \neq \vec{0}$, $k\vec{u}$ a la même
  direction et sens contraire de $\vec{u}$ et a pour norme $-k||\vec{u}||$ 
  \item [$\bullet$] si $k=0$ ou $\vec{u}=\vec{0}$, alors $k\vec{u}=\vec{0}$
\end{itemize}
\end{minipage}
&
\begin{minipage}{6cm}
\includegraphics[width=3cm]{images/ex22.png}
\end{minipage}
\end{tabular} 




\subsection{Règles de calcul}

Pour tous vecteurs $\vec{u}$ et $\vec{v}$ et pour tous réels $k$ et $k'$, on a:

\begin{itemize}
  \item [$\bullet$] $k(\vec{u}+\vec{v})=k\vec{u}+k\vec{v}$
  \item [$\bullet$] $(k+k')\vec{u}=k\vec{u}+k'\vec{u}$
  \item [$\bullet$] $k(k'\vec{u})=(k \times k')\vec{u}$
  \item [$\bullet$] $k\vec{u}=\vec{o}$ si et seulement si $k=0$ ou
  $\vec{u}=\vec{0}$
\end{itemize}

\textit{Ex 5, 6 et 7} 

\subsection{Colinéarité}

\ul{Définition}: On dit que deux vecteurs $\vec{u}$ et
$\vec{v}$ sont \textbf{colinéaires} lorsqu'il existe un réel $k$  tel que
$\vec{u}=k\vec{v}$.

\enskip

\ul{Remarque}: Le vecteur nul est colinéaire à tout vecteur $\vec{u}$ car
$\vec{0}=0\vec{u}$.

\enskip

\ul{Propriété}: Deux vecteurs non nuls $\vec{u}$ et $\vec{v}$ colinéaires ont
même direction.

\subsection{Repère du plan}

\ul{Rappels}: Trois points $O$, $I$ et $J$ non-alignés forment un repère. Cela
revient au même de se donner $O$ et deux vecteurs $\vec{i}$ et $\vec{j}$
\textbf{non colinéaires} tels que: $\overrightarrow{OI}=\vec{i}$ et
$\overrightarrow{OJ}=\vec{j}$. Un repère du plan est donc indifféremment la
donnée des triplets $(O, I, J)$ ou $(O, \vec{i}, \vec{j})$. $O$ est l'origine
du repère.

\enskip

\ul{Définition}: On dit que le couple $(\vec{i},\vec{j})$ est une \textbf{base}
 du plan.

\subsection{Colinéarité}



\ul{Propriété}: $\vec{u} \begin{pmatrix} x\\ y \end{pmatrix}$ et $\vec{v}
\begin{pmatrix} x' \\ y' \end{pmatrix}$ sont colinéiares si et seulement si:

\begin{itemize}

  $x$ & $x'$ \\
  \hline 
  $y$ & $y'$\\
  \hline
  \end{tabular}  est un tableau de proportionnalité. 
\end{itemize}

\begin{pmatrix} -21\\ 14 \end{pmatrix}$ sont-ils colinéaires?

\subsection{Norme et distance entre deux points}


\ul{Propriété}: Soient $A(x_A;y_A)$ et $B(x_B;y_B)$ deux points quelconques du
plan  et $(O, \vec{i}, \vec{j})$ un repère \textbf{orthonormal}. La distance de
$A$ à $B$ est donnée par $AB= \sqrt{(x_B-x_A)^2+(y_B-y_A)^2}$. 

\enskip

\ul{Conséquence}: Dans une base \textbf{orthonormale} $(\vec{i}, \vec{j})$, la
norme d'un vecteur $\vec{u} \begin{pmatrix} x\\ y \end{pmatrix}$ est donnée par
                            
                            
$||\vec{u}||=\sqrt{x^2+y^2}$.

\section{Equations de droites}

Soit un repère $(O, \vec{i}, \vec{j})$.

\subsection{Equations réduites, équations cartésiennes}

\ul{Définition}: Soit $A$ un point du plan et $\vec{u}$ un vecteur non nul.
L'ensemble de tous les points $M$ du plan tel que $\vec{u}$ et
$\overrightarrow{AM}$ sont colinéaires est une droite. $\vec{u}$ est appelé
\textbf{vecteur directeur}.

\enskip

\ul{Remarques}: Pour tous points distincts $A$ et $B$, $\overrightarrow{AB}$ est
un vecteur directeur de la droite $(AB)$.

\enskip


\ul{Propriété (Rappel)}: Toute droite du plan a une équation de la forme:
\begin{itemize}
  \item [$\bullet$] $x=c$ si la droite est parallèle à l'axe des ordonnées
  \item [$\bullet$] $y=ax+b$ sinon.
\end{itemize}


\enskip

\ul{Démonstration}: Soit $A \begin{pmatrix} x_A\\ y_A \end{pmatrix}$ et $B
\begin{pmatrix} x_B\\ y_B \end{pmatrix}$ deux points distincts. Un point $M
\begin{pmatrix} x\\ y \end{pmatrix}$ appartient à la droite $(AB)$ si et
seulement si $\overrightarrow{AB}$ et $\overrightarrow{AM}$ sont colinéaires.


$\overrightarrow{AB} \begin{pmatrix} \ldots \ldots \ldots\\ \ldots \ldots \ldots
\end{pmatrix}$ et $\overrightarrow{AM} \begin{pmatrix} \ldots \ldots \ldots\\
\ldots \ldots \ldots \end{pmatrix}$ 

Donc Un point $M \begin{pmatrix} x\\ y \end{pmatrix}$ appartient à la droite
$(AB)$ si et seulement si $(.. \ldots \ldots)(.. \ldots \ldots)=(..
\ldots \ldots)(\ldots \ldots ..)$

c'est-à-dire si et seulement si: $(x-x_A)(y_B-y_A)-(y-y_A)(x_B-x_A)=0$.


\enskip

\textbf{Cas 1:} Si $x_B-x_A=0$

On a alors $(AB)$ parallèle à l'axe des ordonnées.

Donc: $(x-x_A)(y_B-y_A)=0$ c'est-à-dire \ldots \ldots \ldots \ldots ou \ldots
\ldots \ldots \ldots

\begin{center}
      \fbox{$x=x_A$}
      \end{center}


\textbf{Cas 2:} Si $x_B-x_A \neq 0$


$(y-y_A)(x_B-x_A)=(x-x_A)(y_B-y_A)$

$y=\dfrac{y_B-y_A}{x_B-x_A} (x-x_A)+y_A$

$y=\dfrac{y_B-y_A}{x_B-x_A} x + \dfrac{x_By_A-x_Ay_B}{x_B-x_A}$

\begin{center}
      \fbox{$y=ax+b$}
      \end{center}

\enskip

\ul{Propriété}: Toute équation de droites peut se mettre sous la forme $\alpha
x+ \beta y+ \gamma=0$ appelée \textbf{équation cartésienne} (avec $\alpha$ et
$\beta$ non nuls simultanément).


\enskip


\ul{Démonstration}:
\begin{itemize}
  \item [$\bullet$] $y=ax+b$ peut s'écrire \ldots \ldots \ldots \ldots \ldots
  
  et $x=c$ peut s'écrire \ldots \ldots \ldots \ldots \ldots
  \item [$\bullet$] Réciproquement si l'équation est de la forme: $\alpha
x+ \beta y+ \gamma=0$  alors:

si $\beta=0$, $\alpha x + \gamma=0$ donc $x=\dfrac{- \gamma}{\alpha}$
(car $\alpha$ et $\beta$ non nuls simultanément).

 
  si $\beta \neq 0$, $\beta y= -\alpha x -\gamma$ donc
  $y=\ldots\ldots\ldots\ldots\ldots$
\end{itemize}

\ul{Théorème}: Soit $\vec{u}$ un vecteur directeur d'une droite $(d)$ de
coordonnées $\begin{pmatrix} \alpha\\ \beta \end{pmatrix}$ dans une base
$(\vec{i},\vec{j})$, c'est-à-dire tel que $\vec{u}=\alpha \vec{i}+\beta
\vec{j}$ avec $\alpha$ et $\beta$ non nuls à la fois.

\begin{itemize}
  \item [$\bullet$] Si $\alpha =0$, alors $\vec{u}=\beta \vec{j}$ avec $\beta
  \neq 0$ et donc $(d)$ est parallèle à l'axe des ordonnées.
  \item [$\bullet$] Si $\alpha \neq 0$, alors $(d)$ est sécante à l'axe des
  ordonnées et son coefficient directeur est $a=\dfrac{\beta}{\alpha}$.
\end{itemize}
