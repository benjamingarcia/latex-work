\section*{\center{Bilan intervalles}}




\bigskip
\begin{center}
\begin{tabular}{|m{9cm}|m{10cm}|}
\hline
\textbf{Ce que je dois savoir} & \textbf{Ce que je dois savoir faire} \\
\hline
\begin{itemize}
  \item[$\bullet$] Je connais la définition d'un intervalle et je comprends le
  sens des crochets.
  \item [$\bullet$] Je connais la définition d'une union et d'une intersection
  d'intervalles.
\end{itemize}
&

\enskip
\begin{itemize}
  \item[$\bullet$] Je sais reconnaître un intervalle à partir d'un encadrement
  ou d'une représentation sur la droite graduée.
  \item[$\bullet$] Je peux représenter la droite graduée et écrire des
  encadrements à partir d'un intervalle.
  \item[$\bullet$] Je suis capable de déterminer l'intersection de plu
 sieurs intervales.
 \item[$\bullet$] je suis capable de déterminer l'union de plusieurs
 intervalles. \end{itemize} \\
\hline

\end{tabular}
\end{center}
