\section*{\center{Propriétés des droites parallèles et perpendiculaires}}

\bigskip
\bigskip

\textbf{Exercice 1:}

\begin{enumerate}
  
  \item Construire une droite $(d)$ et placer deux points $A$ et $B$ qui
  n'appartiennent pas à la droite $(d)$.
  \item Construire en rouge la droite $(d_{1})$ parallèle à $(d)$ passant par
  $A$.
  \item Construire en vert la droite $(d_{2})$ parallèle à $(d)$ passant par
  $B$.
  \item  Que peut-on dire des droites $(d_{1})$ et $(d_{2})$?
  \item Recopier et compléter la phrase: \textit{Les droites $(d_{1})$ et
  $(d_{2})$ sont \ldots \ldots \ldots \ldots \ldots parce qu'elles sont toutes les deux
  \ldots\ldots\ldots\ldots\ldots\ldots à la droite (\ldots).}
  \item Citer la propriété du cours qui permet de l'affirmer?
 \item Recopier et compléter:
 \begin{center}
 
\begin{tabular}{cc}
 \begin{minipage}{6cm}
 \ul{Données}\\
 \ldots \ldots \ldots \ldots \ldots\\
 \ldots \ldots \ldots \ldots \ldots
 \end{minipage}       
&  
\begin{minipage}{6cm}
\ul{Conclusion}
\medskip

donc \ldots \ldots \ldots \ldots
\end{minipage}
 \end{tabular}
  \end{center}
  \end{enumerate}

\bigskip

\textbf{Exercice 2:}

\begin{tabular}{cc}
\begin{minipage}{9cm}
\includegraphics[width=9cm]{images/ex2.png}
\end{minipage}
&
\begin{minipage}{9cm}
Compléter le texte suivant:


On sait que les droites \ldots et \ldots sont \ldots \ldots \ldots \ldots.

 Or si deux droites sont parallèles à une même droite alors
elles sont \ldots \ldots \ldots \ldots entre elles. 

Donc on a:  \ldots et \ldots
sont parallèles.
\end{minipage}
\end{tabular}

\bigskip 

\textbf{Exercice 3:}

\begin{tabular}{cc}
\begin{minipage}{12cm}
\includegraphics[width=12cm]{images/paralleles.jpg}
\end{minipage}
&
\begin{minipage}{4cm}
Sur cette figure, on a:


\enskip


$(d_{1})  //  (d_{4})$\\
\quad $(d_{3})  //  (d_{2})$\\
\quad $(d_{5})  //  (d_{4})$\\
\quad $(d_{5}) //  (d_{3})$\\
\quad $(\Delta_{1})  //  (\Delta_{5})$\\
\quad $(\Delta_{3})  //  (\Delta_{4})$\\
\quad $(\Delta_{1})  //  (\Delta_{4})$\\
\quad $(\Delta_{2})  //  (\Delta_{4})$\\

\end{minipage}
\end{tabular}


\begin{enumerate}
  \item Montrer que les droites $(d_{1})$ et $(d_{5})$ sont parallèles.
  \item Montrer que les droites $(\Delta_{4})$ et $(\Delta_{5})$ sont
  parallèles.
\end{enumerate}
