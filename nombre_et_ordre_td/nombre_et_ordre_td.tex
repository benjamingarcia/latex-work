\section*{\center{TD nombre et ordre}}

\textbf{Définition}: Soit $a$ et $b$ deux réels. $a>b$ ou ($b<a$) signifie que
$a-b>0$. De même, $a \geqslant b $ ou ($b \leqslant a$) signifie qua $a-b
\geqslant 0$.

\bigskip

\textbf{Exercice 1:} Classer les deux séries de nombres dans l'ordre
croissant:
\begin{enumerate}
  \item \enskip 1,02 \qquad 10,2 \qquad 1,0203 \qquad 1,020 \qquad 1,2 \qquad
  1,23
  \item \enskip -2,3 \qquad -2,5 \qquad -1,02 \qquad -1,021
\end{enumerate}

\bigskip
\textbf{Exercice 2}: 
\begin{enumerate}
  \item  Compléter par le signe $<$ ou $>$:\\ $\dfrac{1}{4} \ldots
  \dfrac{1}{2}$;  \qquad $\dfrac{1}{7} \ldots \dfrac{1}{8}$; \qquad
  $\dfrac{1}{5} \ldots \dfrac{1}{55}$; \qquad $\dfrac{1}{3} \ldots
  \dfrac{1}{\pi}$;
  \item  A partir de ces exemples, compléter la propriété suivante:\\
Soit $a$ et $b$ deux réels avec $a>b$, $a>0$ et $b>0$ alors $\dfrac{1}{a} \ldots
\dfrac{1}{b}$.
\item Trouver un contre exemple qui montre l'importance de l'hypothèse $a>0$ et
$b>0$.
\end{enumerate}

\bigskip
\textbf{Exercice 3}: Compléter les propriétés suivantes par $<$ ou $>$ (vous
pouvez d'abord regarder ce qu'il se passe sur des exemples):
\begin{enumerate}
  \item Soit $a,b,c$ trois réels avec $a>b$ alors $a+c \ldots b+c$.
  \item Soit $a,b,c$ trois réels avec $a>b$ alors $a-c \ldots b-c$.
\end{enumerate}

\bigskip

\textbf{Exercice 4}: 
\begin{enumerate}
  \item Compléter la propriété suivante:\\
Soit $a,b,c$ trois réels avec $a>b$  et $c>0$ alors $ac \ldots bc$ et
$\dfrac{a}{c} \ldots \dfrac{b}{c}$.
  \item Avec les mêmes hypothèses sur $a$ et $b$, que se passe t'il lorsque
  $c<0$? Enoncer la propriété.
\end{enumerate}


\bigskip

\textbf{Exercice 5}: Comparer sans calcul et donner la règle utilisée:
\begin{enumerate}
  \item $\dfrac{2}{7}$ et $\dfrac{5}{7}$
  \item $\dfrac{5}{7}$ et $\dfrac{15}{11}$
  \item $\dfrac{-1}{3}$ et $\dfrac{5}{6}$
  \item $\dfrac{4}{7}$ et $\dfrac{4}{9}$
  \item $\dfrac{7}{8}$ et $1,2$
\end{enumerate}

\bigskip
\textbf{Exercice 6}: Compléter par le symbole $<$ ou $>$ (sans faire de calcul):
\begin{enumerate}
  \item $-7+\sqrt{2} \ldots -3+\sqrt{2}$
  \item $\dfrac{-7}{\pi} \ldots \dfrac{-3}{\pi}$
  \item $-7-10^{7} \ldots -3-10^{7}$
  \item $-7 \times (-5,2) \ldots -3 \times (-5,2)$
  \item $-7 \times \dfrac{1}{4} \ldots -3 \times \dfrac{1}{4}$
  \item $-7:(-1) \ldots -3:(-1)$
\end{enumerate}
