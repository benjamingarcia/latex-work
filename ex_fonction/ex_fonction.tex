\textbf{Exercice 6:} 
\begin{enumerate}
  \item A l'aide de votre calculatrice, conjecturer l'existence d'un extremum
  des fonctions suivantes: $f(x)=x^{2}-7$ \qquad \quad $g(x)=-x^{2}+5$ \qquad
  \quad $h(x)=6+(x-5)^{2}$
  \item Démontrer vos conjectures.
\end{enumerate}

\textbf{Exercice 7:}
\begin{enumerate}
  \item Montrer que la fonction $f$ définie sur $\mathbb{R}$ par $f(x)=5x-100$
  est croissante sur $\mathbb{R}$.
 \item Montrer que la fonction $g$ définie sur $\mathbb{R}$ par $g(x)=-2x+3$
  est décroissante sur $\mathbb{R}$.  
\end{enumerate}
