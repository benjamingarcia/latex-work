\section*{\center{Bilan droites remarquables du triangle}}




\bigskip

\begin{center}
\begin{tabular}{|m{9cm}|m{9cm}|}
\hline


\textbf{Ce que je dois savoir} & \textbf{Ce que je dois savoir faire} \\

\hline

\enskip


\begin{itemize}
  \item [$\bullet$] Je connais la définition de médiatrice, médiane, hauteur et
  bissectrice.
  
  
  \item [$\bullet$] Je sais reconnaître une médiatrice, une médiane, une
  hauteur ou une bissectrice grâce au codage.
  
  \item [$\bullet$] Je connais la propriété du cercle circonscrit d'un triangle.
  
  \item [$\bullet$] Je sais trouver le centre du cercle circonscrit.
  
  \item [$\bullet$] Je connais la formule de l'aire d'un triangle. 
  
  \item [$\bullet$] Je sais reconnaître la hauteur relative à un côté.

 \end{itemize}

&


\begin{itemize}
  \item[$\bullet$] Je sais tracer une médiane et une hauteur dans un triangle.


  \item[$\bullet$] Je sais tracer la médiatrice d'un segment à l'équerre ou au
  compas.
      
  \item[$\bullet$] Je sais construire la bissectrice d'un angle avec le
  rapporteur ou au compas.
  
  \item[$\bullet$] Je sais tracer le cercle circonscrit d'un triangle.
  
  \item[$\bullet$] Je sais calculer l'aire d'un triangle.
 
\end{itemize} \\

\hline

\end{tabular}
\end{center}
