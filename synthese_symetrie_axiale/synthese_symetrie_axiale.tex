\center{\textbf{\Large{Bilan symétrie axiale}}}




\bigskip

\begin{center}
\begin{tabular}{|m{9cm}|m{10cm}|}
\hline
\textbf{Ce que je dois savoir} & \textbf{Ce que je dois savoir faire} \\
\hline

\enskip


\begin{itemize}
  \item[$\bullet$] Je sais reconnaître si deux figures sont symétriques par
  rapport à une droite.
  \item [$\bullet$] Je connais le vocabulaire.
  \item [$\bullet$] Je connais les propriétés des figures usuelles (axes de
  symétrie).
  \item [$\bullet$] Je connais les propriétés de la symétrie axiale (rayon du
  cercle, longueur d'un segment, mesure d'angles, périmètres et aires d'une
  figure). 
\end{itemize}
&

\enskip
\begin{itemize}
  \item[$\bullet$] Je sais construire le symétrique d'un point par rapport à
  une droite (avec ou sans quadrillage).
  \item[$\bullet$] Je sais construire le symétrique d'une figure par rapport à
  une droite (avec ou sans quadrillage).
  \item[$\bullet$] Je sais construire le symétrique d'une droite, d'un angle,
  d'un segment , ou d'un cercle par rapport à une droite.
 \item[$\bullet$] Je sais reconnaître les axes de symétrie d'une figure.
    \end{itemize} \\
\hline 

\end{tabular}
\end{center}
