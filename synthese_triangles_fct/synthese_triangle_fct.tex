\section*{\center{Bilan fonctions}}




\bigskip
\begin{center}
\begin{tabular}{|m{9cm}|m{10cm}|}
\hline
\textbf{Ce que je dois savoir} & \textbf{Ce que je dois savoir faire} \\
\hline
\begin{itemize}
  \item[$\bullet$] Je connais le vocabulaire.
  \item [$\bullet$] Je sais utiliser la calculatrice pour les calculs et les
  graphiques.
  \item [$\bullet$] Je comprends les différents aspects de la notion de
  fonction: graphiques, calculs, tableaux (de valeurs, de signe et de
  variation).
\end{itemize}
&

\enskip
\begin{itemize}
  \item[$\bullet$] Je sais reconnaître le graphe d'une fonction.
  \item[$\bullet$] J'identifie la variable et le domaine de définition de la
  fonction.
  \item[$\bullet$] Je sais remplir et je comprends un tableau de valeurs.
 \item[$\bullet$] Je sais lire un tableau de variation, je sais le compléter
 et le décrire avec le vocabulaire adapté.
  \item[$\bullet$] Je sais calculer des images et des antécédents.
  \item[$\bullet$] Je sais interpréter la représentation graphique d'une
  fonction: image, antécédent, minimum, maximum, monotonie
  (croissante/décroissante), signe de la fonction.
  \item[$\bullet$] Je sais résoudre graphiquement des équations et  inéquations.
 
 \end{itemize} \\
\hline

\end{tabular}
\end{center}

\section*{\center{Bilan triangles isométriques et semblables}}




\bigskip
\begin{center}
\begin{tabular}{|m{9cm}|m{10cm}|}
\hline
\textbf{Ce que je dois savoir} & \textbf{Ce que je dois savoir faire} \\
\hline
\begin{itemize}
  \item[$\bullet$] Je connais les définitions et propriétés des transformations
  usuelles.
  \item [$\bullet$] Je connais la définition de triangles isométriques. 
  \item [$\bullet$] Je connais les cas d'isométries de triangles.
  \item [$\bullet$] Je connais la propriété d'isométrie dans le cas d'un
  triangle rectangle.
  \item [$\bullet$] Je connais la définition de triangles semblables (de mêmes
  formes).
  \item [$\bullet$] Je connais les cas de similitude de triangles.
\end{itemize}
&

\enskip
\begin{itemize}
  \item[$\bullet$] Je sais identifier les sommets homologues de triangles
  isométriques.
  \item[$\bullet$] Je suis capable de déterminer des égalités de longueur et
  d'angle sur des triangles isométriques.
  \item[$\bullet$] Je sais identifier les sommets homologues de triangles
  semblables.
 \item[$\bullet$] Je sais écrire des égalités de rapport dans deux triangles
 semblables et des égalités d'angle.
  \item[$\bullet$] Je sais calculer le rapport des aires de deux triangles
  semblables.
 \end{itemize} \\
\hline

\end{tabular}
\end{center}
