\center{\textbf{\Large{Bilan tableaux et graphiques}}}




\bigskip

\begin{center}
\begin{tabular}{|m{9cm}|m{10cm}|}
\hline
\textbf{Ce que je dois savoir} & \textbf{Ce que je dois savoir faire} \\
\hline

\enskip


\begin{itemize}
  \item[$\bullet$] Je sais lire un tableau à deux lignes.
  \item [$\bullet$] Je sais lire un tableau à deux colonnes.
  \item [$\bullet$] Je sais lire un tableau à double entrée.
  \item [$\bullet$] Je sais lire et interpréter un graphique.
  \item [$\bullet$] Je connais le vocabulaire: abscisse, ordonnée.
\end{itemize}
&

\enskip
\begin{itemize}
  \item[$\bullet$] Je sais compléter un tableau à deux lignes ou à deux
  colonnes.
  \item[$\bullet$] Je sais construire un tableau à deux lignes ou à deux
  colonnes.
  \item[$\bullet$] Je sais compléter un tableau à double entrée.
 \item[$\bullet$] Je sais construire un tableau à double entrée.
 \item[$\bullet$] Je sais construire un graphique à partir d'un tableau.
    \end{itemize} \\
\hline 

\end{tabular}
\end{center}
