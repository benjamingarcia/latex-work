\textbf{\ul{Correction exercice 23 p 122}}

\end{center}

\textbf{Triangle SRT}:

\begin{itemize}
  \item [$\bullet$] Tracer [ST] d'une longueur de 7cm.
  \item [$\bullet$] Construire l'angle $\widehat{S}$ mesurant 70°. Prolonger la
  demi-droite obtenue.
  \item [$\bullet$] Tracer [SR] d'une longueur de 7cm (le point R appartient à
  la demi-droite précédente).
  \item [$\bullet$] Tracer [RT].
  
\end{itemize}

\enskip

\textbf{Triangle RTU}:

\begin{itemize}
  \item [$\bullet$] Construire le point U à l'aide du compas en prenant comme
  écartement la longueur RT.
  \item [$\bullet$] Tracer [RU] et [TU].
   \end{itemize}


\enskip


\textbf{Triangle VSR}:

\begin{itemize}
  \item [$\bullet$] Tracer la penpendiculaire à la droite (RS) passant par R.
  On la nomme ($d_1$).
  \item [$\bullet$] Tracer la perpendiculaire à la droite (ST) passant par S.
  On la nomme ($d_2$).
  \item [$\bullet$] Placer le point V qui est le point d'intersection de la
  droite ($d_1$) et de la droite ($d_2$).
  
\end{itemize}

\enskip


$\bullet$ \textbf{Mettre le codage.}
