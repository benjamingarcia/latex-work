\bigskip

\hline

  \item [$\bullet$] Je connais les limites de ma calculatrice

  \item[$\bullet$] Je sais écrire un nombre en écriture scientifique

 \end{itemize} \\

\subsection*{Valeurs absolues}


\bigskip
\begin{center}
\begin{tabular}{|p{9cm}|p{9cm}|}
\hline
\textbf{Ce que je dois savoir} & \textbf{Ce que je dois savoir faire} \\
\hline
\begin{itemize}
  \item[$\bullet$] Je connais la notation $|\ldots|$ et sa définition
  \item [$\bullet$] Je sais représenter graphiquement des solutions d'équations
  ou d'inéquations avec des valeurs absolues
  \item [$\bullet$] Je sais interpréter et traduire des équations/inéquations
  avec valeurs absolues en terme de distance
  \item [$\bullet$] Je connais la formule $\sqrt{a^{2}}=|a|$
\end{itemize}
&

\enskip
\begin{itemize}
  \item[$\bullet$] Je sais calculer une valeur absolue
  \item[$\bullet$] Je sais résoudre des équations et des inéquations avec des
  valeurs absolues
  \end{itemize} \\
\hline

\end{tabular}
\end{center}

  n'est pas démontrer.
  \item [$\bullet$] \textbf{Voir} sur une figure n'est pas démontrer.
  \item [$\bullet$] \textbf{Conjecturer} c'est deviner, imaginer, émettre des
  hypothèses.
  \item [$\bullet$] \textbf{Construire} c'est dessiner, en justifiant par des
  définitions et/ou des propriétés, la figure exécutée.
\end{itemize}
