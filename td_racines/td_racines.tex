\section*{\center{TD Racines}}


\subsection*{Rappels}

\textbf{Définition:} $a$ étant un nombre \textbf{positif} (ou nul), $\sqrt{a}$
est le nombre positif (ou nul) qui élevé au carré donne $a$:  \quad $\sqrt{a}
\times \sqrt{a}=(\sqrt{a})^2=a$ avec $a\geqslant 0$.
 
 \bigskip
 \textbf{Règles:}
 \begin{enumerate}
   \item \fbox{$\sqrt{a}$ n'est définie que si $a$ est un nombre
   \textbf{positif} (ou nul).}
  \medskip 
\item $a$ étant un nombre positif, il existe $2$ nombres, $\sqrt{a}$ et
$-\sqrt{a}$ qui élevés au carré donnent $a$.
\item $a$ et $b$ étant des nombres positifs et $b \not =0$, on a:\\
$\sqrt{a} \times \sqrt{b}=\sqrt{ab}$\\ 
$\sqrt{a^{2}}=a$ car $a \geqslant 0$\\ 
$\dfrac{\sqrt{a}}{\sqrt{b}}=\sqrt{\dfrac{a}{b}}$.
 \end{enumerate}
 
 \textit{ATTENTION:} il faut calculer le nombre sous le radical $\sqrt{\   }$
 \textbf{avant} de calculer la racine carrée!\\
 
 \subsection*{Applications}
 
 
 \textbf{Exercice 1:} ($\diamond$) Calculer $\sqrt{a+b}$ et $\sqrt{a}+\sqrt{b}$
 pour:\\
 $a=1 $ et $b=3$ \\
 $a=9$ et $b=16$\\
 Que remarque t'on?
 
 
 \bigskip
 \textbf{Exercice 2:} ($\diamond$) Ecrire les nombres suivants sous forme
 $a\sqrt{b}$ où $a$ et $b$ sont des entiers naturels et $b$ le plus petit
 possible: \\
 $\sqrt{27}$, \thinspace $\sqrt{200}$,  \thinspace $\sqrt{8}$, \thinspace
 $\sqrt{75}$, \thinspace $\sqrt{18}$.
 
 
 \bigskip
 \textbf{Exercice 3:} ($\diamond$) Réduire au maximum les nombres
 suivants:
 \begin{enumerate}
   \item [1)]$2 \sqrt{3} - 4 \sqrt{3} + \sqrt{27}$
   \item [2)]$\sqrt{45} + 3 \sqrt{5} - 3 \sqrt{20}$ 
   \item [3)]$\dfrac{3  \sqrt{5} \times \sqrt{12}}{2 \sqrt{15}}$
   \item [4)]$(\sqrt{3})^{2}- (\dfrac{2 \sqrt{2}}{3})^{2}$
 \end{enumerate}
