\section*{\center{Bilan fonction affine}}




\bigskip

\begin{center}
\begin{tabular}{|m{9cm}|m{9cm}|}
\hline


\textbf{Ce que je dois savoir} & \textbf{Ce que je dois savoir faire} \\

\hline

\enskip


\begin{itemize}
  
  
  \item [$\bullet$] Je sais reconnaitre l'expression algébrique d'une fonction
  affine.
  
  \item [$\bullet$] Je sais reconnaitre graphiquement une fonction affine.

\item [$\bullet$] Je connais le vocabulaire: ordonnée à l'origine, coefficient
directeur, fonction constante.

\item [$\bullet$] Je connais la différence entre fonction affine et fonction
linéaire.

\item [$\bullet$] Je connais la formule permettant de retrouver le coefficient
directeur d'une fonction affine.

 \end{itemize}

&


\begin{itemize}
  
  
  \item [$\bullet$] Je sais calculer image et antécédent d'un nombre par une
  fonction affine.
  
  
\item[$\bullet$] Je sais tracer la représentation graphique d'une fonction
affine.

  \item[$\bullet$] Je sais lire graphiquement l'ordonnée à l'origine et le
  coefficient directeur d'une fonction affine.

\item[$\bullet$] Je sais trouver l'expression algébrique d'une fonction
  affine par le calcul.
 
\item[$\bullet$] Je sais trouver l'expression algébrique d'une fonction
  affine graphiquement.
  
  \item[$\bullet$] Je sais résoudre des problèmes.
 

\end{itemize} \\

\hline

\end{tabular}
\end{center}
