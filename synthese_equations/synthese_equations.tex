\section*{\center{Bilan équations}}




\bigskip

\begin{center}
\begin{tabular}{|m{10cm}|m{9cm}|}
\hline


\textbf{Ce que je dois savoir} & \textbf{Ce que je dois savoir faire} \\

\hline

\enskip


\begin{itemize}
  \item [$\bullet$] Je connais le vocabulaire: équation, inconnue de
  l'équation, solution de l'équation, membre de droite, membre de gauche et
  résoudre une équation.
  
  
  \item [$\bullet$] Je sais qu'additionner ou soustraire un même nombre à
  chaque membre donne une autre équation (ayant les mêmes solutions).
  
  \item [$\bullet$] Je sais que multiplier ou diviser un même nombre (non nul) à
  chaque membre donne une autre équation (ayant les mêmes solutions).


 \end{itemize}

&


\begin{itemize}
  \item[$\bullet$] Je sais vérifier si un nombre est solution d'une équation.


\item[$\bullet$] Je sais résoudre des problèmes simples.

\item[$\bullet$] Je sais résoudre une équation.
\item[]
  \item[$\bullet$] Je sais mettre un problème en équation.

 
\end{itemize} \\

\hline

\end{tabular}
\end{center}
