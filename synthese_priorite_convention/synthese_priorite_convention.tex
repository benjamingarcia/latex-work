\section*{\center{Bilan enchainements d'opérations sur les nombres}}




\bigskip

\begin{center}
\begin{tabular}{|m{9cm}|m{9cm}|}
\hline
\textbf{Ce que je dois savoir} & \textbf{Ce que je dois savoir faire} \\
\hline

\enskip


\begin{itemize}
  \item[$\bullet$] Je connais les règles de priorité de calculs.
  
\enskip

  \item [$\bullet$] Je connais la signification de $a^2$ et $a^3$ et je sais les
  calculer sur des exemples.
  
  \enskip
  
  \item [$\bullet$] Je sais rajouter les signes ``$\times$'' qui sont
  sous-entendus.

\enskip

  \item [$\bullet$] Je peux supprimer un signe ``$\times$'' devant une lettre
  ou une parenthèse.
 
 \enskip
  
  \item [$\bullet$] Je connais le vocabulaire associé aux différentes
  opérations.
 
 \enskip
  
  \item [$\bullet$] Je connais la méthode pour calculer une expression
  littérale pour une certaine valeur de lettres.
  
 
\end{itemize}
&

\enskip
\begin{itemize}
  \item[$\bullet$] Je sais effectuer des calculs en respectant les priorités.

\enskip

  \item[$\bullet$] Je sais effectuer des calculs avec des expressions
  fractionnaires.

 
\enskip  
      
  \item[$\bullet$] Je sais reconnaître si une expression est une somme, un
  produit, un quotient ou une différence.
  
\enskip
  
 \item[$\bullet$] A partir d'une phrase utilisant le vocabulaire des
 opérations, je sais écrire le calcul correspondant.
 
\enskip
 
  \item[$\bullet$] Je sais résoudre un problème et écrire l'expression
  permettant de résoudre le problème. 
  

 \end{itemize} \\
\hline

\end{tabular}
\end{center}
