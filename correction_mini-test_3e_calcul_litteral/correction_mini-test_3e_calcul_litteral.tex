\ul{Exercice 6}: 

\begin{enumerate}
  \item \textit{$-4$ est-il solution de l'équation $4v-9=-25$? }
  
  
  $4 \times v -9=4 \times (-4)-9=-16-9=-25$ \qquad (-4) est bien solution de
  l'équation.



\enskip

\item \textit{$7$ est-il solution de l'équation $7b-1=6+4b$?}
  
  
  $7 \times b -1=7 \times 7 -1=49-1=48$ \qquad et \qquad $6+4\times b=6+4
  \times 7 = 6+28=34$
  
  $48 \neq 34$ donc 7 n'est pas solution de l'équation.
\end{enumerate}

\ul{Exercice 6}: 

\begin{enumerate}
  \item \textit{$-3$ est-il solution de l'équation $4d-10=-11$?}
  

$4 \times d -10 = 4 \times (-3)-10=-12-10=-22$ \qquad $-22 \neq =-11$ donc (-3)
n'est pas solution de l'équation.


\enskip

\item \textit{$5$ est-il solution de l'équation $5v-3=2+4v$?}
  
  
  
$5 \times v-3=5 \times 5-3=25-3=22$ \qquad et \qquad $2 + 4 \times v=2+4 \times
5=2+20=22$


donc 5 est solution de l'équation.
\end{enumerate}
