\center{\textbf{\Large{Bilan symétrie centrale et axiale}}}




\bigskip

\begin{center}
\begin{tabular}{|m{9cm}|m{10cm}|}
\hline
\textbf{Ce que je dois savoir} & \textbf{Ce que je dois savoir faire} \\
\hline

\enskip


\begin{itemize}
  \item[$\bullet$] Je connais la définition d'un point symétrique par rapport à
  un autre.
  \item [$\bullet$] Si A et B sont symétriques par rapport à O, je sais placer
  le point O.
  \item [$\bullet$] Je connais les propriétés de la symétrie centrale
  (longueur de segments, mesure d'angle, rayon de cercles , droites parallèles,
  périmètres et aires d'une figure).
  \item [$\bullet$] Je connais la définition et les propriétés de
  la symétrie axiale.
  \item [$\bullet$] Je connais la différence entre symétrie axiale (miroir) et
  symétrie centrale (demi-tour) et je sais les reconnaître.
  \item [$\bullet$] Je connais la méthode pour tracer le symétrique d'une
  droite par rapport  à un axe.
\end{itemize}
&

\enskip
\begin{itemize}
  \item[$\bullet$] Je sais construire le symétrique d'un point A par rapport à
  un point O.
  \item[$\bullet$] Je reconnais deux points symétriques par rapport à un
  point O.
  \item[$\bullet$] Je sais tracer le symétrique d'un segment par rapport à
  un point O.
 \item[$\bullet$] Je sais tracer le symétrique d'un cercle par rapport à
  un point O.
  \item[$\bullet$] Je sais tracer le symétrique d'une droite par rapport à
  un point O. 
   \item[$\bullet$] Je sais tracer le symétrique d'une figure par rapport à
  un point O avec les instruments ou en s'aidant d'un quadrillage.
  \item[$\bullet$] Je sais reconnaître si une figure a un axe de symétrie ou un
  centre de symétrie et je sais les placer.
  \item[$\bullet$] Je sais tracer le symétrique d'une figure par rapport à un
  axe.
  \item[$\bullet$] Je sais utiliser les propriétés des deux symétries pour
 justifier un résultat.
  
    \end{itemize} \\
\hline 

\end{tabular}
\end{center}
